import java.util.*;
class Prog_11_Extract_words_from_sentences
{
    public static void main()
    {
        Scanner sc=new Scanner(System.in);
        String str,w="";
        char ch;
        int i,len;
        System.out.println("Enter a sentence");
        str=sc.nextLine();
        str=str+" ";
        len=str.length();
        for(i=0;i<len;i++)
        {
            ch=str.charAt(i);
            if(ch!=' ')// when the extracted character is not a space
            {
                w=w+ch;
            }
            else// when the character is a space
            {
                System.out.println(w);
                w="";
            }
        }
    }

import java.util.*;
class Prog_12_Largest_Word
{
    public static void main(String args[])
    {
        Scanner sc=new Scanner(System.in);
        String str,w="",lw="";
        char ch;
        int len,i,llw=0,l;
        System.out.println("Enter a string");
        str=sc.nextLine();
        str=str+" ";
        len=str.length();
        for(i=0;i<len;i++)
        {
            ch=str.charAt(i);
            if(ch!=' ')
            {
                w=w+ch;
            }
            else
            {
                l=w.length();
                if(l>llw)
                {
                    llw=l;
                    lw=w;
                }
                w="";
            }
        }
        System.out.println("The largest word is :"+lw);
        System.out.println("The length of the largest word is :"+llw);
    }
}

 *  step 8 : store llw as lenth of accumulated word if the word length > llw
 *  step 9 : store the length of largest word

 *  step 12: print lenth of largest word

        {
            ch=str.charAt(i);

import java.util.*;
class BinarySearch
{
    public static void main()
    {
        Scanner in = new Scanner(System.in);
        int arr[]=new int[10];
        int lb,ub,mid=0,sv,i,k=0 ;
        System.out.println("Enter numbers in Ascending order");
        for(i=0;i<10;i++)
        {
            System.out.println("Enter a number");
            arr[i]=in.nextInt();
        }
        System.out.println("Enter a number to be searched");
        sv=in.nextInt();
        lb=0;
        ub=9;
        while(lb<=ub)
        {
            mid=(lb+ub)/2;
            if(arr[mid]==sv)
            {
                k=1;
                break;
            }
            if (arr[mid]>sv)
            {
                ub=mid-1;
            }
            if(arr[mid]<sv)
            {
                lb=mid+1;
            }
        }
        if(k==0)
        {
            System.out.println("Value not found");
        }
        else
        {
            System.out.println("Value found at "+(mid+1) +" position");
        }
    }
}
/*
    Step 1 : Calculate the mid element of the array.
    Step 2 : Compare the value  with the mid element.
    Step 3 : If key = middle element, then we return the mid index position for the value found.
    Step 4 : Else If key > mid element, then the key lies in the right half of the array. 
    Step 5 : repeat steps 1 to 3 on the lower (right) half of the array.
    Step 6 : Else key < mid element, then the value is in the upper half of the array.
    Step 7 : repeat the binary search in the upper half.
 * 
 */
/* variable     type    description
 * arr[]        array   to store values in the array
 * ub           int     to store the index of max value in the array
 * lb           int     to store the index of min value in the array
 * i            int     loop variable
 * mid          int     to store the index of mid value of the array
 * sv           int     to store the search value
 * k            int     indicates a value is present or not 
 */

import java.util.*;
class BubbleSort
{
    public static void main()
    {
        Scanner in = new Scanner(System.in);
        int n=5;
        int arr[]=new int[n];
        int i , j ,t;
        for(i=0;i<n;i++)
        {
            System.out.println("Enter a number");
            arr[i]=in.nextInt();
        }
        System.out.println("Original values");
        for(i=0;i<n;i++)
        {
            System.out.print(arr[i]+" , ");
        }
        System.out.println();
        //bubble sort
        for(i=0;i<n;i++)
        {
            for(j=0;j<n-1-i;j++)
            {
                if(arr[j]>arr[j+1])
                {
                    t=arr[j];
                    arr[j]=arr[j+1];
                    arr[j+1]=t;
                }
            }
        }
        System.out.println("values after sort");
        for(i=0;i<n;i++)
        {
            System.out.print(arr[i]+" , ");
        }
    }
}

 * Step 6 : repeat step 3 and step 5
 * Step 7 : print the sorted array
 */
